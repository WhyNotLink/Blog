# 前言

小车主控芯片选用MSPM0G4507，因为嘉立创送开发板，所以选用天猛星开发板。

天猛星重要引脚：PA19（SWDOIO）、PA20（SWCLK）、A10(TX)、A11（RX）

基于天猛星我画了一套小车底板，包含Mpu6050(陀螺仪)、OLED(0.96寸IIC单色屏)、超声波、LED（四个）、灰度传感器（8通道）、按键（2个）、蜂鸣器、tb6612（2个，最多支持控制小车四个轮子）、A4988（2个，最多支持控制两个步进电机）

电源是12V电池输入，然后内部转为5V和3.3。底座也提供了相应的拓展插座

基于此我觉得应该是完全够用了，具体设计可以看立创分享的文件，本文我会把所有设备驱动代码示例全写一遍

由于是TI的芯片，所以软件上要用到Keil和SYSCONFIG，（我习惯用Keil所以没有用他的TICCS）利用SYSCONFIG生成库函数然后通过Keil调试。

# KEY

在syscfg里面（本文不讲如何使用这个软件）GPIO生成一个name为KEY1，KEY2，然后找个定时器1ms中断

### KEY.h

```
#ifndef _KEY_H_
#define _KEY_H_
 
#include "ti_msp_dl_config.h"
#include "stdbool.h"
 
struct key
{
    bool state;
    int casenum;
    int setflag;
};

#endif
```

### KEY.c

```
#include "KEY.h"
volatile struct key keys[2];

void TIMER_KEY_INST_IRQHandler(void)
{
	//读取io口电平
    {	
			
				keys[0].state=DL_GPIO_readPins(GPIOA,KEY1_PIN_0_PIN);
				keys[1].state=DL_GPIO_readPins(GPIOA,KEY2_PIN_1_PIN);
    }	
    //io口电平处理
    for(int i=0;i<2;i++)
    {
        switch(keys[i].casenum)
        {
					//case 0 用来检测电平是否有变化，如果有就将casenum+1
            case 0:
                {
                    if(keys[i].state==0)
                        {
                            keys[i].casenum=1;
                        }
                }
                break;
					//case 1 用来检测电平是否真的被按下，如果有则casenum+1,如果只是虚假的按下(电平波动造成)casenum-1
            case 1:
                {
                    if(keys[i].state==0)
                    {
                        keys[i].casenum=2;
                    }
                    else
                    {
                        keys[i].casenum=0;
                    }
                }
                break;
					//松手判定的标志位，在main文件中读取keys[i].state是否为1就可以判断是否被按下
            case 2:
                {
                    if(keys[i].state==1)
                    {
                        keys[i].setflag=1;
                        keys[i].casenum=0;
                    }
                }
                break;
        }
    }

}
```

```
//电平拉低LED导通亮，拉高灭
void LED_off(int num)
{
    switch (num)
    {
    case 1:
    		DL_GPIO_setPins(LED1_PORT,LED1_PIN_22_PIN); 
        break;
    case 2:
			DL_GPIO_setPins(LED2_PORT,LED2_PIN_30_PIN);

        break;
    case 3:
			DL_GPIO_setPins(LED3_PORT,LED3_PIN_0_PIN);

        break;
    case 4:
			DL_GPIO_setPins(LED4_PORT,LED4_PIN_1_PIN);
        break;
		default:
			break;
    }
}
 
void LED_on(int num)
{
    switch (num)
    {
    case 1:
			DL_GPIO_clearPins(LED1_PORT,LED1_PIN_22_PIN);
        break;
    case 2:
			DL_GPIO_clearPins(LED2_PORT,LED2_PIN_30_PIN);
        break;
    case 3:
			DL_GPIO_clearPins(LED3_PORT,LED3_PIN_0_PIN);
        break;
    case 4:
			DL_GPIO_clearPins(LED4_PORT,LED4_PIN_1_PIN);
        break;
		default:
			break;
    }
}
```

```
void OLED_WR_Byte(u8 dat,u8 mode)
{
    uint16_t i;

    uint8_t Send_Buff[5] = {0};

    uint8_t SendData_Count = 2;

    if(mode)
        Send_Buff[0] = 0x40;
    else
        Send_Buff[0] = 0x00;

    Send_Buff[1] = dat;

    DL_I2C_fillControllerTXFIFO(I2C_OLED_INST, Send_Buff, SendData_Count);

    while (!(DL_I2C_getControllerStatus(I2C_OLED_INST) & DL_I2C_CONTROLLER_STATUS_IDLE));
    DL_I2C_startControllerTransfer(I2C_OLED_INST, 0x3C, DL_I2C_CONTROLLER_DIRECTION_TX, SendData_Count);

}

//更新显存到OLED	
void OLED_Refresh(void)
{
	u8 i,n;
	for(i=0;i<8;i++)
	{
		OLED_WR_Byte(0xb0+i,OLED_CMD); //设置行起始地址
		OLED_WR_Byte(0x00,OLED_CMD);   //设置低列起始地址
		OLED_WR_Byte(0x10,OLED_CMD);   //设置高列起始地址

        OLED_WR_Byte(0x40,OLED_DATA);
		for(n=0;n<128;n++)
		{
			OLED_WR_Byte(OLED_GRAM[n][i], OLED_DATA);
		}
    }
}

//画线
//x1,y1:起点坐标
//x2,y2:结束坐标
void OLED_DrawLine(u8 x1,u8 y1,u8 x2,u8 y2,u8 mode)
{
	u16 t; 
	int xerr=0,yerr=0,delta_x,delta_y,distance;
	int incx,incy,uRow,uCol;
	delta_x=x2-x1; //计算坐标增量 
	delta_y=y2-y1;
	uRow=x1;//画线起点坐标
	uCol=y1;
	if(delta_x>0)incx=1; //设置单步方向 
	else if (delta_x==0)incx=0;//垂直线 
	else {incx=-1;delta_x=-delta_x;}
	if(delta_y>0)incy=1;
	else if (delta_y==0)incy=0;//水平线 
	else {incy=-1;delta_y=-delta_x;}
	if(delta_x>delta_y)distance=delta_x; //选取基本增量坐标轴 
	else distance=delta_y;
	for(t=0;t<distance+1;t++)
	{
		OLED_DrawPoint(uRow,uCol,mode);//画点
		xerr+=delta_x;
		yerr+=delta_y;
		if(xerr>distance)
		{
			xerr-=distance;
			uRow+=incx;
		}
		if(yerr>distance)
		{
			yerr-=distance;
			uCol+=incy;
		}
	}
}

void OLED_DrawCircle(u8 x,u8 y,u8 r)
{
	int a, b,num;
    a = 0;
    b = r;
    while(2 * b * b >= r * r)      
    {
        OLED_DrawPoint(x + a, y - b,1);
        OLED_DrawPoint(x - a, y - b,1);
        OLED_DrawPoint(x - a, y + b,1);
        OLED_DrawPoint(x + a, y + b,1);
 
        OLED_DrawPoint(x + b, y + a,1);
        OLED_DrawPoint(x + b, y - a,1);
        OLED_DrawPoint(x - b, y - a,1);
        OLED_DrawPoint(x - b, y + a,1);
        
        a++;
        num = (a * a + b * b) - r*r;//计算画的点离圆心的距离
        if(num > 0)
        {
            b--;
            a--;
        }
    }
}

//mode:0,反色显示;1,正常显示
void OLED_ShowChar(u8 x,u8 y,u8 chr,u8 size1,u8 mode)
{
	u8 i,m,temp,size2,chr1;
	u8 x0=x,y0=y;
	if(size1==8)size2=6;
	else size2=(size1/8+((size1%8)?1:0))*(size1/2);  //得到字体一个字符对应点阵集所占的字节数
	chr1=chr-' ';  //计算偏移后的值
	for(i=0;i<size2;i++)
	{
		if(size1==8)
			  {temp=asc2_0806[chr1][i];} //调用0806字体
		else if(size1==12)
        {temp=asc2_1206[chr1][i];} //调用1206字体
		else if(size1==16)
        {temp=asc2_1608[chr1][i];} //调用1608字体
		else if(size1==24)
        {temp=asc2_2412[chr1][i];} //调用2412字体
		else return;
		for(m=0;m<8;m++)
		{
			if(temp&0x01)OLED_DrawPoint(x,y,mode);
			else OLED_DrawPoint(x,y,!mode);
			temp>>=1;
			y++;
		}
		x++;
		if((size1!=8)&&((x-x0)==size1/2))
		{x=x0;y0=y0+8;}
		y=y0;
  }
}

//显示汉字
//x,y:起点坐标
//num:汉字对应的序号
//mode:0,反色显示;1,正常显示
void OLED_ShowChinese(u8 x,u8 y,u8 num,u8 size1,u8 mode)
{
	u8 m,temp;
	u8 x0=x,y0=y;
	u16 i,size3=(size1/8+((size1%8)?1:0))*size1;  //得到字体一个字符对应点阵集所占的字节数
	for(i=0;i<size3;i++)
	{
		if(size1==16)
				{temp=Hzk1[num][i];}//调用16*16字体
		else if(size1==24)
				{temp=Hzk2[num][i];}//调用24*24字体
		else if(size1==32)       
				{temp=Hzk3[num][i];}//调用32*32字体
		else if(size1==64)
				{temp=Hzk4[num][i];}//调用64*64字体
		else return;
		for(m=0;m<8;m++)
		{
			if(temp&0x01)OLED_DrawPoint(x,y,mode);
			else OLED_DrawPoint(x,y,!mode);
			temp>>=1;
			y++;
		}
		x++;
		if((x-x0)==size1)
		{x=x0;y0=y0+8;}
		y=y0;
	}
}

//num 显示汉字的个数
//space 每一遍显示的间隔
//mode:0,反色显示;1,正常显示
void OLED_ScrollDisplay(u8 num,u8 space,u8 mode)
{
	u8 i,n,t=0,m=0,r;
	while(1)
	{
		if(m==0)
		{
	        OLED_ShowChinese(128,24,t,16,mode); //写入一个汉字保存在OLED_GRAM[][]数组中
			t++;
		}
		if(t==num)
        {
            for(r=0;r<16*space;r++)      //显示间隔
            {
                for(i=1;i<144;i++)
                {
                    for(n=0;n<8;n++)
                    {
                        OLED_GRAM[i-1][n]=OLED_GRAM[i][n];
                    }
                }
                OLED_Refresh();
            }
            t=0;
        }
		m++;
		if(m==16){m=0;}
		for(i=1;i<144;i++)   //实现左移
		{
			for(n=0;n<8;n++)
			{
				OLED_GRAM[i-1][n]=OLED_GRAM[i][n];
			}
		}
		OLED_Refresh();
	}
```
```
}

//起始信号
void I2C_Start(void)
{
	OLED_SDA_Set();
	OLED_SCL_Set();
	IIC_delay();
	OLED_SDA_Clr();
	IIC_delay();
	OLED_SCL_Clr();
	IIC_delay();
}

//结束信号
void I2C_Stop(void)
{
	OLED_SDA_Clr();
	OLED_SCL_Set();
	IIC_delay();
	OLED_SDA_Set();
}

//等待信号响应
void I2C_WaitAck(void) //测数据信号的电平
{
	OLED_SDA_Set();
	IIC_delay();
	OLED_SCL_Set();
	IIC_delay();
	OLED_SCL_Clr();
	IIC_delay();
}

//写入一个字节
void Send_Byte(u8 dat)
{
	u8 i;
	for(i=0;i<8;i++)
	{
		if(dat&0x80)//将dat的8位从最高位依次写入
		{
			OLED_SDA_Set();
    }
		else
		{
			OLED_SDA_Clr();
    }
		IIC_delay();
		OLED_SCL_Set();
		IIC_delay();
		OLED_SCL_Clr();//将时钟信号设置为低电平
		dat<<=1;
  }
}

//发送一个字节

	I2C_Start();
	Send_Byte(0x78);
	I2C_WaitAck();
	if(mode){Send_Byte(0x40);}
  else{Send_Byte(0x00);}
	I2C_WaitAck();
	Send_Byte(dat);
	I2C_WaitAck();
	I2C_Stop();
}

//更新显存到OLED	
void OLED_Refresh(void)
{
	u8 i,n;
	for(i=0;i<8;i++)
	{
		OLED_WR_Byte(0xb0+i,OLED_CMD); //设置行起始地址
		OLED_WR_Byte(0x00,OLED_CMD);   //设置低列起始地址
		OLED_WR_Byte(0x10,OLED_CMD);   //设置高列起始地址
		I2C_Start();
		Send_Byte(0x78);
		I2C_WaitAck();
		Send_Byte(0x40);
		I2C_WaitAck();
		for(n=0;n<128;n++)
		{
			Send_Byte(OLED_GRAM[n][i]);
			I2C_WaitAck();
		}
		I2C_Stop();
  }
}

			 OLED_GRAM[n][i]=0;//清除所有数据

}
```

```
static A4988_Config_t gConfig[3];              // 存储电机配置
static volatile int32_t gTotalSteps[3] = {0};  // 存储累计脉冲数

/**
 * @brief 初始化方向引脚
 */
void A4988_Init(void) {
    DL_GPIO_clearPins(PTZ1_DIR_GPIO_PORT, PTZ1_DIR_PIN);
    DL_GPIO_clearPins(PTZ2_DIR_GPIO_PORT, PTZ2_DIR_PIN);
}

/**
 * @brief 设置电机旋转方向
 * @param id 电机ID (1或2)
 * @param dir 方向 (0或1)
 */
void A4988_SetDir(uint8_t id, uint8_t dir) {
    if (id == 1) {
        if (dir == 0) {
            DL_GPIO_clearPins(PTZ1_DIR_GPIO_PORT, PTZ1_DIR_PIN);
        } else {
            DL_GPIO_setPins(PTZ1_DIR_GPIO_PORT, PTZ1_DIR_PIN);
        }
    } else if (id == 2) {
        if (dir == 0) {
            DL_GPIO_clearPins(PTZ2_DIR_GPIO_PORT, PTZ2_DIR_PIN);
        } else {
            DL_GPIO_setPins(PTZ2_DIR_GPIO_PORT, PTZ2_DIR_PIN);
        }
    }
}

/**
 * @brief 设置PWM频率控制转速 (计算公式: ARR = Clock / Freq - 1)
 */
void A4988_SetSpeed(uint8_t id, uint32_t freq) {
    if (freq < 20) freq = 20; // 限制最低频率防止溢出
    uint32_t arr = 1000000 / freq - 1;

    if (id == 1) {
        DL_TimerA_setLoadValue(PWM_PTZ1_INST, arr);
    } else if (id == 2) {
        DL_TimerA_setLoadValue(PWM_PTZ2_INST, arr);
    }
}

/**
 * @brief 同时设置两个电机的速度
 */
void A4988_SetSpeed_All(uint32_t freq) {
    uint32_t arr = 1000000 / freq - 1;
    DL_TimerA_setLoadValue(PWM_PTZ1_INST, arr);
    DL_TimerA_setLoadValue(PWM_PTZ2_INST, arr);
}

void A4988_Start(void) {
    DL_TimerG_startCounter(PWM_PTZ1_INST);
    DL_TimerG_startCounter(PWM_PTZ2_INST);
}

void A4988_Stop(void) {
    DL_TimerG_stopCounter(PWM_PTZ1_INST);
    DL_TimerG_stopCounter(PWM_PTZ2_INST);
}

void A4988_SetConfig(const A4988_Config_t *config) {
    gConfig[1] = *config;
    gConfig[2] = *config;
}

/**
 * @brief 根据累计步数换算为物理距离(mm)
 */
float A4988_GetDistance_mm(uint8_t id) {
    if (id < 1 || id > 2) return 0.0f;
    
    // 距离 = (当前总脉冲 / 每圈脉冲数) * 每圈行程
    return (float)gTotalSteps[id] * gConfig[id].mm_per_rev / gConfig[id].steps_rev;
}

void A4988_ResetDistance(uint8_t id) {
    if (id >= 1 && id <= 2) {
        gTotalSteps[id] = 0;
    }
}

/* ---------------- 中断服务函数 ---------------- */

/**
 * @brief 电机1脉冲计数 (在定时器周期结束/溢出时触发)
 */
void PWM_PTZ1_INST_IRQHandler(void) {
    // 读取方向引脚状态，决定步数加减
    if (DL_GPIO_readPins(PTZ1_DIR_GPIO_PORT, PTZ1_DIR_PIN) == 0) {
        gTotalSteps[1]++;
    } else {
        gTotalSteps[1]--;
    }
}

 */
void PWM_PTZ2_INST_IRQHandler(void) {
    if (DL_GPIO_readPins(PTZ2_DIR_GPIO_PORT, PTZ2_DIR_PIN) == 0) {
        gTotalSteps[2]++;
    } else {
        gTotalSteps[2]--;
    }
}
```

```
#ifndef _PID_MG513_H
#define _PID_MG513_H

#include "ti_msp_dl_config.h"
#include "tb6612.h"

/* 机械参数配置 */
#define MG513_GEAR_RATIO    30.0f    // 减速比 (根据实际型号修改)
#define MG513_ENCODER_RES   11.0f    // 编码器线数
#define MG513_QEI_MODE      4.0f     // 4倍频开启则为4
#define PULSES_PER_REV      (MG513_GEAR_RATIO * MG513_ENCODER_RES * MG513_QEI_MODE)
#define PID_SAMPLING_TIME_MS  20.0f                            // 采样周期 20ms
#define PID_SAMPLING_FREQ     (1000.0f / PID_SAMPLING_TIME_MS) // 换算成频率 50Hz 

/* QEI 计数器偏移量（与 SysConfig 保持一致） */
#define QEI_OFFSET          32768

/* PID 结构体定义 */
typedef struct {
    float Kp;
    float Ki;
    float Kd;

    float target_speed;   // 目标速度 (单位: 脉冲数/采样周期)
    float current_speed;  // 当前速度
    
    float err;            // 当前误差 e(k)
    float err_last;       // 上次误差 e(k-1)
    float err_prev;       // 上上次误差 e(k-2)

    float output;         // PID 计算后的输出值
    int32_t max_output;   // 输出限幅 (例如 999)
} PID_Control_t;

/* 对外接口 */
void PID_Init(void);
int32_t PID_Speed_Compute(PID_Control_t *pid, float feedback);
float PID_Get_Revolutions(int32_t total_pulses);
void Motor_Update_Task(void); // 放在定时器中断中执行
void Motor_Set_Speed(uint8_t motor_id, float rev_per_sec);
#endif
```

### PID_MG513.c

```
#include "PID_MG513.h"

/* 定义电机 PID 对象 */
PID_Control_t pidA;
PID_Control_t pidB;

/* 累计行程脉冲数 */
static int64_t g_total_pulses_A = 0;
static int64_t g_total_pulses_B = 0;

/**
 * @brief PID 参数初始化
 */
void PID_Init(void) {
    // 电机 A 参数 (需根据负载实际调试)
    pidA.Kp = 2.8f;
    pidA.Ki = 0.5f;
    pidA.Kd = 0.1f;
    pidA.max_output = 999;
    pidA.target_speed = 0;

    // 电机 B 参数
    pidB = pidA; 
}

/**
 * @brief 增量式 PID 计算逻辑
 * @param feedback 采样周期内测得的脉冲变化量
 */
int32_t PID_Speed_Compute(PID_Control_t *pid, float feedback) {
    pid->current_speed = feedback;
    pid->err = pid->target_speed - pid->current_speed;

    // 增量式 PID 公式
    float delta = pid->Kp * (pid->err - pid->err_last) +
                  pid->Ki * pid->err +
                  pid->Kd * (pid->err - 2.0f * pid->err_last + pid->err_prev);

    pid->output += delta;

    // 状态更新
    pid->err_prev = pid->err_last;
    pid->err_last = pid->err;

    // 输出限幅
    if (pid->output > pid->max_output)  pid->output = (float)pid->max_output;
    if (pid->output < -pid->max_output) pid->output = -(float)pid->max_output;

    return (int32_t)pid->output;
}

/**
 * @brief 获取当前累计转过的总圈数
 */
float PID_Get_Revolutions(int32_t total_pulses) {
    return (float)total_pulses / PULSES_PER_REV;
}

/**
 * @brief 电机闭环控制核心任务
 * @note  必须在定时器中断（如 20ms 周期）中调用
 */
void Motor_Update_Task(void) {
    /* 1. 读取编码器 A */
    uint32_t raw_count_A = DL_Timer_getTimerCount(ENCODER_A_INST);
    float speed_A = (float)((int32_t)raw_count_A - QEI_OFFSET);
    g_total_pulses_A += (int32_t)speed_A; // 累加总脉冲
    DL_Timer_setTimerCount(ENCODER_A_INST, QEI_OFFSET); // 重置 QEI

    /* 2. 读取编码器 B */
    uint32_t raw_count_B = DL_Timer_getTimerCount(ENCODER_B_INST);
    float speed_B = (float)((int32_t)raw_count_B - QEI_OFFSET);
    g_total_pulses_B += (int32_t)speed_B;
    DL_Timer_setTimerCount(ENCODER_B_INST, QEI_OFFSET);

    /* 3. PID 计算 */
    int32_t outA = PID_Speed_Compute(&pidA, speed_A);
    int32_t outB = PID_Speed_Compute(&pidB, speed_B);

    /* 4. 执行控制 (TB6612) */
    if (outA >= 0) AO_Control(1, (uint32_t)outA);
    else           AO_Control(0, (uint32_t)(-outA));

    if (outB >= 0) BO_Control(1, (uint32_t)outB);
    else           BO_Control(0, (uint32_t)(-outB));
}

/**
 * @brief 设置电机目标速度
 * @param motor_id 1为A，2为B
 * @param rev_per_sec 目标转速（圈/秒）
 */
void Motor_Set_Speed(uint8_t motor_id, float rev_per_sec) {
    // 换算公式：目标脉冲数 = (目标圈速 * 一圈总脉冲) / 每秒采样次数
    float target = (rev_per_sec * PULSES_PER_REV) / PID_SAMPLING_FREQ;
    
    if (motor_id == 1) {
        pidA.target_speed = target;
    } else if (motor_id == 2) {
        pidB.target_speed = target;
    }
}
```

```
#include "motion_control.h"

/**
 * @brief 内部转换函数：将 圈/秒 转换为 采样周期内的脉冲数
 */
static float Speed_Rev_To_Pulse(float rev_per_sec) {
    // 公式：(目标圈速 * 一圈总脉冲) / 每秒采样次数
    return (rev_per_sec * PULSES_PER_REV) / PID_SAMPLING_FREQ;
}

void Motion_Stop(void) {
    pidA.target_speed = 0.0f;
    pidB.target_speed = 0.0f;
    pidA.output = 0; // 清除累计增量
    pidB.output = 0;
    TB6612_Motor_Stop();
}

void Motion_Straight(float speed_rev) {
    float pulse_target = Speed_Rev_To_Pulse(speed_rev);
    pidA.target_speed = pulse_target;
    pidB.target_speed = pulse_target;
}

void Motion_Turn_In_Place(float speed_rev) {
    float pulse_target = Speed_Rev_To_Pulse(speed_rev);
    // 左轮正，右轮负（根据实际接线可能需要互换）
    pidA.target_speed = pulse_target;
    pidB.target_speed = -pulse_target;
}

void Motion_Set_Differential(float left_speed_rev, float right_speed_rev) {
    pidA.target_speed = Speed_Rev_To_Pulse(left_speed_rev);
    pidB.target_speed = Speed_Rev_To_Pulse(right_speed_rev);
}

MotionState_t Motion_Get_Current_State(void) {
    if (pidA.target_speed == 0 && pidB.target_speed == 0) return STOP;
    if (pidA.target_speed > 0 && pidB.target_speed > 0)   return FORWARD;
    if (pidA.target_speed < 0 && pidB.target_speed < 0)   return BACKWARD;
    if (pidA.target_speed > pidB.target_speed)           return RIGHT_TURN;
    if (pidA.target_speed < pidB.target_speed)           return LEFT_TURN;
    return STOP;
}
```

```
#include "mpu6050.h"
#include "stdio.h"

/******************************************************************
 * 函 数 名 称：MPU6050_WriteReg
 * 函 数 说 明：硬件IIC连续写入数据
 * 函 数 形 参：addr 器件地址，regaddr 寄存器地址，num 长度，regdata 数据地址
 * 函 数 返 回：0 成功
 ******************************************************************/
char MPU6050_WriteReg(uint8_t addr, uint8_t regaddr, uint8_t num, uint8_t *regdata)
{
    // 1. 发送寄存器地址
    // 硬件 I2C 发送起始信号，总长度为 1 (寄存器地址) + num (数据)
    DL_I2C_startTransfer(I2C_MPU6050_INST, addr, DL_I2C_TRANSFER_DIRECTION_TX, num + 1);

    // 发送目标寄存器地址
    while(DL_I2C_isTransmitFIFOFull(I2C_MPU6050_INST));
    DL_I2C_transmitData(I2C_MPU6050_INST, regaddr);

    // 2. 发送后续数据
    for(uint16_t i = 0; i < num; i++)
    {
        while(DL_I2C_isTransmitFIFOFull(I2C_MPU6050_INST));
        DL_I2C_transmitData(I2C_MPU6050_INST, regdata[i]);
    }

    // 3. 等待传输完成
    while(DL_I2C_getRawInterruptStatus(I2C_MPU6050_INST, DL_I2C_INTERRUPT_MASTER_TRANSACTION_COMPLETE) == 0);
    DL_I2C_clearRawInterruptStatus(I2C_MPU6050_INST, DL_I2C_INTERRUPT_MASTER_TRANSACTION_COMPLETE);

    return 0;
}

/******************************************************************
 * 函 数 名 称：MPU6050_ReadData
 * 函 数 说 明：硬件IIC连续读取数据
 ******************************************************************/
char MPU6050_ReadData(uint8_t addr, uint8_t regaddr, uint8_t num, uint8_t* Read)
{
    // 1. 写入要读取的寄存器首地址 (Dummy Write)
    DL_I2C_startTransfer(I2C_MPU6050_INST, addr, DL_I2C_TRANSFER_DIRECTION_TX, 1);
    DL_I2C_transmitData(I2C_MPU6050_INST, regaddr);
    while(DL_I2C_getRawInterruptStatus(I2C_MPU6050_INST, DL_I2C_INTERRUPT_MASTER_TRANSACTION_COMPLETE) == 0);
    DL_I2C_clearRawInterruptStatus(I2C_MPU6050_INST, DL_I2C_INTERRUPT_MASTER_TRANSACTION_COMPLETE);

    // 2. 切换为读取模式
    DL_I2C_startTransfer(I2C_MPU6050_INST, addr, DL_I2C_TRANSFER_DIRECTION_RX, num);
    
    for(uint8_t i = 0; i < num; i++)
    {
        while(DL_I2C_isReceiveFIFOEmpty(I2C_MPU6050_INST));
        Read[i] = DL_I2C_receiveData(I2C_MPU6050_INST);
    }

    // 3. 等待事务结束
    while(DL_I2C_getRawInterruptStatus(I2C_MPU6050_INST, DL_I2C_INTERRUPT_MASTER_TRANSACTION_COMPLETE) == 0);
    DL_I2C_clearRawInterruptStatus(I2C_MPU6050_INST, DL_I2C_INTERRUPT_MASTER_TRANSACTION_COMPLETE);

    return 0;
}

uint8_t MPU_Set_Gyro_Fsr(uint8_t fsr) {
    uint8_t data = fsr << 3;
    return MPU6050_WriteReg(MPU6050_ADDR, MPU6050_RA_GYRO_CONFIG, 1, &data);
}

uint8_t MPU_Set_Accel_Fsr(uint8_t fsr) {

uint8_t MPU_Set_Rate(uint16_t rate) {
    uint8_t data;
    if(rate > 1000) rate = 1000;
    if(rate < 4) rate = 4;
    data = 1000 / rate - 1;
    return MPU6050_WriteReg(MPU6050_ADDR, MPU6050_RA_SMPLRT_DIV, 1, &data);
}

void MPU6050ReadGyro(short *gyroData) {
    uint8_t buf[6];
    if(MPU6050_ReadData(MPU6050_ADDR, MPU6050_GYRO_OUT, 6, buf) == 0) {
        gyroData[0] = (buf[0] << 8) | buf[1];
        gyroData[1] = (buf[2] << 8) | buf[3];
        gyroData[2] = (buf[4] << 8) | buf[5];
    }
}

void MPU6050ReadAcc(short *accData) {
    uint8_t buf[6];
    if(MPU6050_ReadData(MPU6050_ADDR, MPU6050_ACC_OUT, 6, buf) == 0) {
        accData[0] = (buf[0] << 8) | buf[1];
        accData[1] = (buf[2] << 8) | buf[3];
        accData[2] = (buf[4] << 8) | buf[5];
    }
}

float MPU6050_GetTemp(void) {
    uint8_t buf[2];
    short temp;
    MPU6050_ReadData(MPU6050_ADDR, MPU6050_RA_TEMP_OUT_H, 2, buf);
    temp = (buf[0] << 8) | buf[1];
    return (float)temp / 340.0f + 36.53f;
}

uint8_t MPU6050ReadID(void) {
    uint8_t id = 0;
    MPU6050_ReadData(MPU6050_ADDR, MPU6050_WHO_AM_I, 1, &id);
    return (id == 0x68) ? 0 : 1;
}

char MPU6050_Init(void) {
    uint8_t val;

    // 1. 复位器件
    val = 0x80;
    MPU6050_WriteReg(MPU6050_ADDR, MPU6050_RA_PWR_MGMT_1, 1, &val);
    delay_ms(100);

    // 2. 唤醒并设置时钟源 (PLL with X axis gyro reference)
    val = 0x01;
    MPU6050_WriteReg(MPU6050_ADDR, MPU6050_RA_PWR_MGMT_1, 1, &val);

    // 3. 配置量程
    MPU_Set_Gyro_Fsr(3);  // ±2000dps
    MPU_Set_Accel_Fsr(0); // ±2g
    MPU_Set_Rate(50);     // 50Hz采样

    // 4. 最终检查ID
    return MPU6050ReadID();
}
```

```
void IIC_Stop(void)

}

/******************************************************************
 * 函 数 名 称：IIC_Send_Ack
 * 函 数 说 明：主机发送应答或者非应答信号
 * 函 数 形 参：0发送应答  1发送非应答
 * 函 数 返 回：无
 * 作       者：LC
 * 备       注：无
******************************************************************/
void IIC_Send_Ack(unsigned char ack)
{
        SDA_OUT();
        SCL(0);
        SDA(0);
        delay_us(5);
        if(!ack) SDA(0);
        else     SDA(1);
        SCL(1);
        delay_us(5);
        SCL(0);
        SDA(1);
}


/******************************************************************
 * 函 数 名 称：I2C_WaitAck
 * 函 数 说 明：等待从机应答
 * 函 数 形 参：无
 * 函 数 返 回：0有应答  1超时无应答
 * 作       者：LC
 * 备       注：无
******************************************************************/
unsigned char I2C_WaitAck(void)
{

        char ack = 0;
        unsigned char ack_flag = 10;
        SCL(0);
        SDA(1);
        SDA_IN();

        SCL(1);
        while( (SDA_GET()==1) && ( ack_flag ) )
        {
                ack_flag--;
                delay_us(5);
        }

        if( ack_flag <= 0 )
        {
                IIC_Stop();
                return 1;
        }
        else
        {
                SCL(0);
                SDA_OUT();
        }
        return ack;
}

/******************************************************************
 * 函 数 名 称：Send_Byte
 * 函 数 说 明：写入一个字节
 * 函 数 形 参：dat要写人的数据
 * 函 数 返 回：无
 * 作       者：LC
 * 备       注：无
******************************************************************/
void Send_Byte(uint8_t dat)
{
        int i = 0;
        SDA_OUT();
        SCL(0);//拉低时钟开始数据传输

        for( i = 0; i < 8; i++ )
        {
                SDA( (dat & 0x80) >> 7 );
                delay_us(1);
                SCL(1);
                delay_us(5);
                SCL(0);
                delay_us(5);
                dat<<=1;
        }
}

/******************************************************************
 * 函 数 名 称：Read_Byte
 * 函 数 说 明：IIC读时序
 * 函 数 形 参：无
 * 函 数 返 回：读到的数据
 * 作       者：LC
 * 备       注：无
******************************************************************/
unsigned char Read_Byte(void)
{
    unsigned char i,receive=0;
    SDA_IN();//SDA设置为输入
    for(i=0;i<8;i++ )
    {
        SCL(0);
        delay_us(5);
        SCL(1);
        delay_us(5);
        receive<<=1;
        if( SDA_GET() )
        {
            receive|=1;
        }
        delay_us(5);
    }

    SCL(0);

    return receive;
}

/******************************************************************
 * 函 数 名 称：MPU6050_WriteReg
 * 函 数 说 明：IIC连续写入数据
 * 函 数 形 参：addr器件地址 regaddr寄存器地址 num要写入的长度 regdata写入的数据地址
 * 函 数 返 回：0=读取成功   其他=读取失败
 * 作       者：LC
 * 备       注：无
******************************************************************/
char MPU6050_WriteReg(uint8_t addr,uint8_t regaddr,uint8_t num,uint8_t *regdata)
{
    uint16_t i = 0;
    IIC_Start();
    Send_Byte((addr<<1)|0);
    if( I2C_WaitAck() == 1 ) {IIC_Stop();return 1;}
    Send_Byte(regaddr);
    if( I2C_WaitAck() == 1 ) {IIC_Stop();return 2;}

    for(i=0;i<num;i++)
    {
        Send_Byte(regdata[i]);
        if( I2C_WaitAck() == 1 ) {IIC_Stop();return (3+i);}
    }

    IIC_Stop();
    return 0;
}


/******************************************************************
 * 函 数 名 称：MPU6050_ReadData
 * 函 数 说 明：IIC连续读取数据
 * 函 数 形 参：addr器件地址 regaddr寄存器地址 num要读取的长度 Read读取到的数据要存储的地址
 * 函 数 返 回：0=读取成功   其他=读取失败
 * 作       者：LC
 * 备       注：无
******************************************************************/
char MPU6050_ReadData(uint8_t addr, uint8_t regaddr,uint8_t num,uint8_t* Read)
{
        uint8_t i;
        IIC_Start();
        Send_Byte((addr<<1)|0);
        if( I2C_WaitAck() == 1 ) {IIC_Stop();return 1;}
        Send_Byte(regaddr);
        if( I2C_WaitAck() == 1 ) {IIC_Stop();return 2;}

        IIC_Start();
        Send_Byte((addr<<1)|1);
        if( I2C_WaitAck() == 1 ) {IIC_Stop();return 3;}

        for(i=0;i<(num-1);i++){
                Read[i]=Read_Byte();
                IIC_Send_Ack(0);
        }
        Read[i]=Read_Byte();
        IIC_Send_Ack(1);
        IIC_Stop();
        return 0;
}

/******************************************************************
 * 函 数 名 称：MPU6050ReadGyro
 * 函 数 说 明：读取陀螺仪数据
 * 函 数 形 参：陀螺仪数据存储地址
 * 函 数 返 回：无
 * 作       者：LC
 * 备       注：无
******************************************************************/
void MPU6050ReadGyro(short *gyroData)
{
        uint8_t buf[6];
        uint8_t reg = 0;
        //MPU6050_GYRO_OUT = MPU6050陀螺仪数据寄存器地址
        //陀螺仪数据输出寄存器总共由6个寄存器组成，
        //输出X/Y/Z三个轴的陀螺仪传感器数据，高字节在前，低字节在后。
        //每一个轴16位，按顺序为xyz
        reg = MPU6050_ReadData(0x68,MPU6050_GYRO_OUT,6,buf);
        if( reg == 0 )
        {
                gyroData[0] = (buf[0] << 8) | buf[1];
                gyroData[1] = (buf[2] << 8) | buf[3];
                gyroData[2] = (buf[4] << 8) | buf[5];
        }
}

/******************************************************************
 * 函 数 名 称：MPU6050_Init
 * 函 数 说 明：MPU6050初始化
 * 函 数 形 参：无
 * 函 数 返 回：0成功  1没有检测到MPU6050
 * 作       者：LC
 * 备       注：无
******************************************************************/
char MPU6050_Init(void)
{
    SDA_OUT();
    delay_ms(10);
    //复位6050
    MPU6050_WriteReg(0x68,MPU6050_RA_PWR_MGMT_1, 1,(uint8_t*)(0x80));
    delay_ms(100);
    //电源管理寄存器
    //选择X轴陀螺作为参考PLL的时钟源，设置CLKSEL=001
    MPU6050_WriteReg(0x68,MPU6050_RA_PWR_MGMT_1,1, (uint8_t*)(0x00));

    MPU_Set_Gyro_Fsr(3);    //陀螺仪传感器,±2000dps
    MPU_Set_Accel_Fsr(0);   //加速度传感器,±2g
    MPU_Set_Rate(50);

    MPU6050_WriteReg(0x68,MPU_INT_EN_REG , 1,(uint8_t*)0x00);        //关闭所有中断
    MPU6050_WriteReg(0x68,MPU_USER_CTRL_REG,1,(uint8_t*)0x00);       //I2C主模式关闭
    MPU6050_WriteReg(0x68,MPU_FIFO_EN_REG,1,(uint8_t*)0x00);         //关闭FIFO
    MPU6050_WriteReg(0x68,MPU_INTBP_CFG_REG,1,(uint8_t*)0X80);       //INT引脚低电平有效

    if( MPU6050ReadID() == 0 )//检查是否有6050
    {
            MPU6050_WriteReg(0x68,MPU6050_RA_PWR_MGMT_1, 1,(uint8_t*)0x01);//设置CLKSEL,PLL X轴为参考
            MPU6050_WriteReg(0x68,MPU_PWR_MGMT2_REG, 1,(uint8_t*)0x00);//加速度与陀螺仪都工作
            MPU_Set_Rate(50);
            return 1;
    }
    return 0;
}
```
